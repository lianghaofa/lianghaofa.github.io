    public List<Integer> findSmallestSetOfVertices(int n, List<List<Integer>> edges) {
        List<Integer> ans = new ArrayList<>();
        int[] in = new int[n];
        for (List<Integer> list : edges) {
            in[list.get(1)]++;
        }
        for (int i = 0; i < in.length; i ++){
            if (in[i] == 0){
                ans.add(i);
            }
        }
        return ans;
    }
