最小差值 I

leetcode-908-SmallestRangeI

class Solution {
    public int smallestRangeI(int[] nums, int k) {
        int max = Integer.MIN_VALUE;
        int min = Integer.MAX_VALUE;
        for (int num : nums) {
            max = Math.max(max, num);
            min = Math.min(min, num);
        }
        int a = Math.abs(max - min);
        int b = 2 * k;
        if (a > b){
            return a - b;
        }
        return 0;
    }
}
