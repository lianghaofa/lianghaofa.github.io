class Solution {
    public boolean hasGroupsSizeX(int[] deck) {
        int[] count = new int[10000];
        for (int item : deck) {
            count[item]++;
        }
        int g0 = -1;
        for (int value : count) {
            if (value == 0) {
                continue;
            }
            if (g0 == -1) {
                g0 = value;
            } else {
                g0 = gcd(g0, value);
            }
        }
        return g0 >= 2;
    }

    private int gcd(int a, int b){
        if (b == 0){
            return a;
        }
        return gcd(b, a % b);
    }
}
