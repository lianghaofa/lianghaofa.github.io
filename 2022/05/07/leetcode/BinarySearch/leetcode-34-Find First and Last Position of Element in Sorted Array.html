    public int[] searchRange(int[] nums, int target) {
        int l = searchLeft(nums, target);
        int r = searchRight(nums, target);
        return new int[]{l , r};
    }

    private int searchRight(int[] nums, int target) {
        int ans = -1;
        int l = 0;
        int r = nums.length - 1;
        while (l <= r){
            int mid = l + ( r - l >> 1);
            if (nums[mid] == target){
                ans = mid;
                l = mid + 1;
            }else if (nums[mid] > target){
                r = mid - 1;
            }else {
                l = mid + 1;
            }
        }
        return ans;
    }

    private int searchLeft(int[] nums, int target) {
        int ans = -1;
        int l = 0;
        int r = nums.length - 1;
        while (l <= r){
            int mid = l + ( r - l >> 1);
            if (nums[mid] == target){
                ans = mid;
                r = mid - 1;
            }else if (nums[mid] > target){
                r = mid - 1;
            }else {
                l = mid + 1;
            }
        }
        return ans;
    }
