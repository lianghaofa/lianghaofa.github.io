x 的平方根

class Solution {
    public int mySqrt(int x) {
        int ans = 0;
        int l = 0;
        int r = x;
        while (l <= r){
            int mid = l + (r - l >> 1);
            if ((long)mid * mid == x){
                return mid;
            }else if ((long)mid * mid > x){
                r = mid - 1;
            }else {
                ans = mid;
                l = mid + 1;
            }
        }
        return ans;
    }
}
