leetcode-528-RandomPickwithWeight

public class Solution {

    int[] ans;
    int sum = 0;
    public Solution(int[] w) {
        ans = new int[w.length];
        int sum = 0;
        for (int i = 0;i < ans.length; i ++){
            sum += w[i];
            ans[i] = sum;
        }
        this.sum = sum;
    }

    public int pickIndex() {
        int s = (int) (Math.random() * sum) + 1;
        int l = 0;
        int r = ans.length - 1;
        int res = ans.length - 1;
        while (l <= r){
            int mid = l + (r - l >> 1);
            if (ans[mid] == s){
                return mid;
            }else if (ans[mid] > s){
                res = mid;
                r = mid - 1;
            }else {
                l = mid + 1;
            }
        }
        return res;
    }
}
